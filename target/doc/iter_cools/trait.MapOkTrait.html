Создает итератор, который заменяет элементы итерационной последовательности.

use iter_cools::*;    
     
let x = vec![Ok(1),Ok(2),Err("ups"),Ok(4)];
let y = x.iter().cloned().map_ok(|x| x + 2).collect::<Result<Vec<usize>,_>>();
assert_eq!(Err("ups"), y);
